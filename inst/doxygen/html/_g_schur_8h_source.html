26 inline std::vector<int> int2bin(int n, int b = 1) {
27  std::vector<int> s;
28  int n1 = n / b;
29  int dif = n - n1 * b;
30  int m = b;
31  do {
32  if (n1 & 1) {
33  s.push_back(m);
34  }
35  m <<= 1;
36  } while ((n1 >>= 1) > 0);
37  if (dif) {
38  s.insert(s.begin(), dif);
39  }
40  return s;
41 }

261  // fft of alpha0, beta0, eta0, xi0, etat, xit
262  gsr->FFT->fft(gsr->alpha0_fft, gsr->alpha0);
263  gsr->FFT->fft(gsr->beta0_fft, gsr->beta0);
264  gsr->FFT->fft(gsr->eta0_fft, gsr->eta0);
265  gsr->FFT->fft(gsr->xi0_fft, gsr->xi0);
266  if (n - npad) {
267  for (int ii = 0; ii < n; ++ii) {
268  gsr->etat[n - ii] = gsr->eta0[ii];
269  gsr->xit[n - ii] = gsr->xi0[ii];
270  }
271  gsr->FFT->fft(gsr->etat_fft, gsr->etat);
272  gsr->FFT->fft(gsr->xit_fft, gsr->xit);
273  }
274  else {
275  for (int ii = 0; ii < n; ++ii) {
276  gsr->etat_fft[2 * ii] = std::conj(gsr->eta0_fft[2 * ii]);
277  gsr->etat_fft[2 * ii + 1] = -std::conj(gsr->eta0_fft[2 * ii + 1]);
278  gsr->xit_fft[2 * ii] = std::conj(gsr->xi0_fft[2 * ii]);
279  gsr->xit_fft[2 * ii + 1] = -std::conj(gsr->xi0_fft[2 * ii + 1]);
280  }
281  }
282  // elementwise complex products
283  for(int ii=0; ii<nn; ii++) {
284  gsr->alphan_fft[ii] = gsr->alpha0_fft[ii] * gsr->eta0_fft[ii] - gsr->xi0_fft[ii] * gsr->beta0_fft[ii];
285  gsr->betan_fft[ii] = gsr->beta0_fft[ii] * gsr->etat_fft[ii] - gsr->xit_fft[ii] * gsr->alpha0_fft[ii];
286  }

329 inline void GSchurN::prog_step(const double* alpha0, const double* beta0, int n) {
330  double *xi1, *xi2, *eta1, *eta2, *swap_ptr; // some pointers to simplify notation
331  double alpha1, alpha2;
332  // Initialize the memory
333  std::fill(gsb_[0]->xin, gsb_[0]->xin + 2 * n, 0);
334  std::fill(gsb_[0]->etan, gsb_[0]->etan + 2 * n, 0);
335  // assign pointers to members of gsb_[0]
336  eta1 = gsb_[0]->etan;
337  eta2 = gsb_[0]->etan + n;
338  xi1 = gsb_[0]->xin;
339  xi2 = gsb_[0]->xin + n;
340  // initialize algorithm
341  eta1[0] = 1.0;
342  xi1[0] = alpha0[0] / beta0[0];
343  gsb_[0]->gamma[0] = xi1[0];
344  gsb_[0]->beta0[0] = beta0[0] * (1 - xi1[0] * xi1[0]);
345  for (int kk = 1; kk < n; ++kk) {
346  alpha2 = alpha0[kk];
347  gsb_[0]->beta0[kk] = beta0[kk];
348  for (int jj = 1; jj <= kk; ++jj) {
349  alpha1 = alpha2 - gsb_[0]->gamma[jj - 1] * gsb_[0]->beta0[kk - jj + 1];
350  gsb_[0]->beta0[kk - jj + 1] -= gsb_[0]->gamma[jj - 1] * alpha2;
351  alpha2 = alpha1;
352  }
353  gsb_[0]->gamma[kk] = alpha1 / gsb_[0]->beta0[0];
354  gsb_[0]->beta0[0] *= 1 - gsb_[0]->gamma[kk] * gsb_[0]->gamma[kk];
355  eta2[0] = 1.0;
356  xi2[0] = alpha0[0] / beta0[0];
357  for (int jj = 1; jj <= kk; ++jj) {
358  xi2[jj] = xi1[jj] + gsb_[0]->gamma[kk] * eta1[kk - jj];
359  eta2[jj] = eta1[jj] + gsb_[0]->gamma[kk] * xi1[kk - jj];
360  }
361  swap_ptr = xi1;
362  xi1 = xi2;
363  xi2 = swap_ptr;
364  swap_ptr = eta1;
365  eta1 = eta2;
366  eta2 = swap_ptr;
367  }
368  std::copy(xi1, xi1 + n, gsb_[0]->xi2n);
369  std::copy(eta1, eta1 + n, gsb_[0]->eta2n);
370  return;
371 }

379 inline void GSchurN::recur_step(const double* alpha0, const double* beta0, int n, int layer) {
380  if (n <= bmod_) {
381  // for small n use prog_step only
382  prog_step(alpha0, beta0, n);
383  return;
384  }
385  prog_step(alpha0, beta0, bmod_);
386  int n_ = bmod_;
387  for(int mm = 0; mm < layer; mm++) {
388  // Prepare the following inputs for compute_nn:
389  // alpha0, beta0, xi0, eta0, and gamma
390  std::copy(alpha0, alpha0 + 2 * n_, gsb_[mm + 1]->alpha0);
391  std::copy(beta0, beta0 + 2 * n_, gsb_[mm + 1]->beta0);
392  std::copy(gsb_[mm]->eta2n, gsb_[mm]->eta2n + n_, gsb_[mm + 1]->eta0);
393  std::copy(gsb_[mm]->xi2n, gsb_[mm]->xi2n + n_, gsb_[mm + 1]->xi0);
394  std::copy(gsb_[mm]->gamma, gsb_[mm]->gamma + n_, gsb_[mm + 1]->gamma);
395  // Compute alphan and betan
396  compute_nn(gsb_[mm + 1], n_, n_);
397  // Compute etan, xin and gamman using recur_step
398  recur_step(gsb_[mm + 1]->alphan + n_, gsb_[mm + 1]->betan + n_, n_, mm);
399  // Prepare the inputs for compute_2n, including xin and etan
400  std::copy(gsb_[mm]->xi2n, gsb_[mm]->xi2n + n_, gsb_[mm + 1]->xin);
401  std::fill(gsb_[mm + 1]->xin + n_, gsb_[mm + 1]->xin + 2 * n_, 0);
402  std::copy(gsb_[mm]->eta2n, gsb_[mm]->eta2n + n_, gsb_[mm + 1]->etan);
403  std::fill(gsb_[mm + 1]->etan + n_, gsb_[mm + 1]->etan + 2 * n_, 0);
404  // Compute eta2n and xi2n
405  compute_2n(gsb_[mm + 1], 2 * n_);
406  // Append gamman after gamma0 to get gamma2n
407  std::copy(gsb_[mm]->gamma, gsb_[mm]->gamma + n_, gsb_[mm + 1]->gamma + n_);
408  n_ <<= 1;
409  }
410  return;
411 }

414 inline void GSchurN::merge_step() {
415  // calculate each piece of GSchur to a different layer of gsm_
416  int layer = log2(ceil((double)sbin_[0] / bmod_));
417  recur_step(alpha_, beta_, sbin_[0], layer);
418  if (nbin_ == 1) {
419  // When vector `sbin_` has only one element, directly pass the results to output.
420  std::copy(gsb_[layer]->eta2n, gsb_[layer]->eta2n + sbin_[0], gsm_[0]->eta2n);
421  std::copy(gsb_[layer]->xi2n, gsb_[layer]->xi2n + sbin_[0], gsm_[0]->xi2n);
422  std::copy(gsb_[layer]->gamma, gsb_[layer]->gamma + sbin_[0], gsm_[0]->gamma);
423  return;
424  }
425  int n_ = 0;
426  for(int mm = 0; mm < nbin_ - 1; mm++) {
427  n_ += sbin_[mm];
428  if(mm) {
429  std::copy(gsm_[mm - 1]->eta2n, gsm_[mm - 1]->eta2n + n_, gsm_[mm]->eta0);
430  std::copy(gsm_[mm - 1]->xi2n, gsm_[mm - 1]->xi2n + n_, gsm_[mm]->xi0);
431  }
432  else {
433  std::copy(gsb_[layer]->eta2n, gsb_[layer]->eta2n + n_, gsm_[mm]->eta0);
434  std::copy(gsb_[layer]->xi2n, gsb_[layer]->xi2n + n_, gsm_[mm]->xi0);
435  std::copy(gsb_[layer]->gamma, gsb_[layer]->gamma + n_, gsm_[nbin_ - 2]->gamma);
436  }
437  // First paste alpha_{0,n} and beta_{0,n} into gsm_[0]->alpha and
438  // gsm_[0]->beta, length is n_
439  std::copy(alpha_, alpha_ + n_ + sbin_[mm + 1], gsm_[mm]->alpha0);
440  std::copy(beta_, beta_ + n_ + sbin_[mm + 1], gsm_[mm]->beta0);
441  // compute_nn uses gsm_[mm]->alpha0, beta0, eta0 and xi0
442  // to compute alphan and betan
443  compute_nn(gsm_[mm], n_, sbin_[mm + 1]);
444  // Computation of recur_step happens inside gsb_[0,1,...,layer].
445  // We just need alphan [sbin_[mm]+(1:sbin_[mm+1])]
446  layer = log2(ceil((double)sbin_[mm + 1] / bmod_));
447  recur_step(gsm_[mm]->alphan + n_, gsm_[mm]->betan + n_, sbin_[mm + 1], layer);
448  std::copy(gsb_[layer]->eta2n, gsb_[layer]->eta2n + sbin_[mm + 1], gsm_[mm]->etan);
449  std::fill(gsm_[mm]->etan + sbin_[mm + 1], gsm_[mm]->etan + n_ + sbin_[mm + 1], 0);
450  std::copy(gsb_[layer]->xi2n, gsb_[layer]->xi2n + sbin_[mm + 1], gsm_[mm]->xin);
451  std::fill(gsm_[mm]->xin + sbin_[mm + 1], gsm_[mm]->xin + n_ + sbin_[mm + 1], 0);
452  std::copy(gsb_[layer]->gamma, gsb_[layer]->gamma + sbin_[mm + 1], gsm_[nbin_ - 2]->gamma + n_);
453  // uses xi_t_fft, xi_0_fft, xi_n_fft (eta), returns xi_0_ifft
454  compute_2n(gsm_[mm], sbin_[mm + 1] + n_);
455  }
456 }

463 inline void GSchurN::compute(double* delta, double& ldV, const double* acf) {
464  // generate alpha0 and beta0 from acf.
465  for (int ii = 0; ii < N_ - 1; ++ii) {
466  alpha_[ii] = -1 * acf[ii + 1];
467  beta_[ii] = acf[ii];
468  }
469  // run gschur algorithm
470  merge_step();
471  // convert gamma into ldV.
472  double sigma2 = log(acf[0]);
473  ldV = sigma2;
474  int k2 = nbin_ > 2 ? nbin_ : 2;
475  for (int ii = 0; ii < N_ - 1; ++ii) {
476  if (gsm_[k2 - 2]->gamma[ii] < 1) {
477  sigma2 += log(1 - gsm_[k2 - 2]->gamma[ii] * gsm_[k2 - 2]->gamma[ii]);
478  ldV += sigma2;
479  }
480  }
481  sigma2 = exp(sigma2);
482  // convert eta and xi into delta, the first column of inverse Toeplitz matrix.
483  std::copy(gsm_[k2 - 2]->eta2n, gsm_[k2 - 2]->eta2n + N_ - 1, delta);
484  delta[N_ - 1] = 0;
485  delta[0] /= sigma2;
486  for (int ii = 1; ii < N_; ++ii) {
487  delta[ii] += gsm_[k2 - 2]->xi2n[ii - 1];
488  delta[ii] /= sigma2;
489  }
490  return;
491 }
