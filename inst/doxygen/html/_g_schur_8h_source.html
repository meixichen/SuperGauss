21 inline std::vector<int> int2Bin(int x, int b = 1) {
22  std::vector<int> s;
23  int x1 = x / b;
24  int dif = x - x1 * b;
25  int n = b;
26  do {
27  if (x1 & 1) {
28  s.push_back(n);
29  }
30  n <<= 1;
31  } while ((x1 >>= 1) > 0);
32  std::reverse(s.begin(), s.end());
33  if (dif) {
34  s.push_back(dif);
35  }
36  return s;
37 }

190  N = N_;
191  b = b_;
192  alpha = new double[N - 1];
193  beta = new double[N - 1];
194  Phi = new double[N];
195 
196  // `s` = \f$ [2^{k_0} \times b, 2^{k_1} \times b, ..., 2^{k_m} \times b, r]
197  // \f$, \f$ k_0 > k_1 > ... > k_m \geq 1 \f$, \f$b \geq r\f$.
198  s = int2Bin(N - 1, b);
199  // \f$k_0\f$ from vector `s`.
200  int k0 = log2(ceil((double)s[0] / b));
201 
202  int gs_size = b;
203  // Necessary memory for \f$T_{,2^{k_0} \times b}\f$ is \f${b, 2\times b,
204  // 4\times b, ..., 2^{k_0}\times b}\f$
205  gs = new GSchur2K*[k0 + 1];
206  gs[0] = new GSchur2K(gs_size << 1);

208  gs[ii + 1] = new GSchur2K(gs_size << 1);
209  gs_size <<= 1;
210  }
211  // Necessary memory for \f$T_{0,N}\f$ is \f${r, r+2^{k_m} \times b, r+2^{k_m}
212  // \times b+2^{k_{m-1}} \times b, ..., r+2^{k_m} \times b+2^{k_{m-1}} \times b

214  // implementation)
215  gsM = new GSchur2K*[s.size()];
216  for (int ii = 0; ii < s.size(); ++ii) {
217  gsM[ii] = new GSchur2K(

219  // b+2^{k_{m-1}} \times b, ..., 2\times 2^{k_0} \times b}\f$ here, which

332 inline void GSchurN::ProgStep(double* alpha0, double* beta0, int n) {
333  double *tmpPtr, *xi1, *xi2, *eta1, *eta2, alpha_0, alpha_1;
334 
335  // Initialize the memory
336  std::fill(gs[0]->xi_n_FFT->in, gs[0]->xi_n_FFT->in + 2 * n, 0);
337  std::fill(gs[0]->eta_n_FFT->in, gs[0]->eta_n_FFT->in + 2 * n, 0);
338  // Take advantage of the memory of existing `GSchur2K` class
339  eta1 = gs[0]->eta_n_FFT->in;
340  eta2 = gs[0]->eta_n_FFT->in + n;
341  xi1 = gs[0]->xi_n_FFT->in;
342  xi2 = gs[0]->xi_n_FFT->in + n;
343 
344  eta1[0] = 1.0;
345  xi1[0] = alpha0[0] / beta0[0];
346  gs[0]->gamma[0] = xi1[0];
347  gs[0]->beta_0_FFT->in[0] = beta0[0] * (1 - xi1[0] * xi1[0]);
348 
349  for (int kk = 1; kk < n; ++kk) {
350  alpha_1 = alpha0[kk];
351  gs[0]->beta_0_FFT->in[kk] = beta0[kk];
352  for (int jj = 1; jj <= kk; ++jj) {
353  alpha_0 =
354  alpha_1 - gs[0]->gamma[jj - 1] * gs[0]->beta_0_FFT->in[kk - jj + 1];
355  gs[0]->beta_0_FFT->in[kk - jj + 1] -= gs[0]->gamma[jj - 1] * alpha_1;
356  alpha_1 = alpha_0;
357  }
358  gs[0]->gamma[kk] = alpha_0 / gs[0]->beta_0_FFT->in[0];
359  gs[0]->beta_0_FFT->in[0] *= 1 - gs[0]->gamma[kk] * gs[0]->gamma[kk];
360 
361  eta2[0] = 1.0;
362  xi2[0] = alpha0[0] / beta0[0];
363  for (int jj = 1; jj <= kk; ++jj) {
364  xi2[jj] = xi1[jj] + gs[0]->gamma[kk] * eta1[kk - jj];
365  eta2[jj] = eta1[jj] + gs[0]->gamma[kk] * xi1[kk - jj];
366  }
367  tmpPtr = xi1;
368  xi1 = xi2;
369  xi2 = tmpPtr;
370  tmpPtr = eta1;
371  eta1 = eta2;
372  eta2 = tmpPtr;
373  }
374  std::copy(xi1, xi1 + n, gs[0]->xi_0_IFFT->out);
375  std::copy(eta1, eta1 + n, gs[0]->eta_0_IFFT->out);
376  return;
377 }

395 inline void GSchurN::GenStep(double* alpha0, double* beta0, int si, int layer) {
396  // When s[i] equals r.
397  if (si <= b) {
398  ProgStep(alpha0, beta0, si < b ? si : b);
399  return;
400  }
401 
402  // When s[i] equals 2^layer * b.
403  ProgStep(alpha0, beta0, b);
404  int n = b;
405  for (int m = 0; m < layer; ++m) {
406  // Prepare the inputs for function alpha2Beta, including alpha-0n, beta-0n,
407  // xi-0n, eta-0n and gamma-0n
408  std::copy(alpha0, alpha0 + 2 * n, gs[m + 1]->alpha_0_FFT->in);
409  std::copy(beta0, beta0 + 2 * n, gs[m + 1]->beta_0_FFT->in);
410  std::copy(gs[m]->gamma, gs[m]->gamma + n, gs[m + 1]->gamma);
411  std::copy(gs[m]->eta_0_IFFT->out, gs[m]->eta_0_IFFT->out + n,
412  gs[m + 1]->eta_0_FFT->in);
413  std::copy(gs[m]->xi_0_IFFT->out, gs[m]->xi_0_IFFT->out + n,
414  gs[m + 1]->xi_0_FFT->in);
415  // Compute alpha-nn and beta-nn
416  alpha2Beta(gs[m + 1], n);
417  // Compute eta-nn, xi-nn and gamma-nn using function GenStep
418  GenStep(gs[m + 1]->alpha_n_IFFT->out + n, gs[m + 1]->beta_n_IFFT->out + n,
419  n, m);
420  // Prepare the inputs for function eta2Xi, including xi-nn and eta-nn
421  std::copy(gs[m]->xi_0_IFFT->out, gs[m]->xi_0_IFFT->out + n,
422  gs[m + 1]->xi_n_FFT->in);
423  std::copy(gs[m]->eta_0_IFFT->out, gs[m]->eta_0_IFFT->out + n,
424  gs[m + 1]->eta_n_FFT->in);
425  // Compute eta-02n and xi-02n
426  eta2Xi(gs[m + 1], n);
427  // Paste the computed gamma-nn after gamma-0n to get gamma-02n
428  std::copy(gs[m]->gamma, gs[m]->gamma + n, gs[m + 1]->gamma + n);
429  n <<= 1;
430  }
431  return;
432 }

436 inline void GSchurN::GenMerge() {
437  int layer = log2(ceil((double)s[0] / b));
438  GenStep(alpha, beta, s[0], layer);
439 
440  // When vector `s` has only one element, merging step is no more necessary.
441  // Directly pass the results to output.
442  if (s.size() == 1) {
443  std::copy(gs[layer]->eta_0_IFFT->out, gs[layer]->eta_0_IFFT->out + s[0],
444  gsM[0]->eta_0_IFFT->out);
445  std::copy(gs[layer]->xi_0_IFFT->out, gs[layer]->xi_0_IFFT->out + s[0],
446  gsM[0]->xi_0_IFFT->out);
447  std::copy(gs[layer]->gamma, gs[layer]->gamma + s[0], gsM[0]->gamma);
448  return;
449  }
450 
451  // When vector `s` has more than one element, merging step is necessary.
452  std::copy(gs[layer]->eta_0_IFFT->out, gs[layer]->eta_0_IFFT->out + s[0],
453  gsM[0]->eta_0_FFT->in);
454  std::copy(gs[layer]->xi_0_IFFT->out, gs[layer]->xi_0_IFFT->out + s[0],
455  gsM[0]->xi_0_FFT->in);
456  std::copy(gs[layer]->gamma, gs[layer]->gamma + s[0], gsM[0]->gamma);
457  int n = s[0];
458  for (int m = 0; m < s.size() - 1; ++m) {
459  if (m == 0) {
460  std::copy(alpha, alpha + N - 1, gsM[m]->alpha_0_FFT->in);
461  std::copy(beta, beta + N - 1, gsM[m]->beta_0_FFT->in);
462  } else {
463  std::copy(gsM[m - 1]->alpha_n_IFFT->out + s[m - 1],
464  gsM[m - 1]->alpha_n_IFFT->out + s[m - 1] + 2 * s[m],
465  gsM[m]->alpha_0_FFT->in);
466  std::copy(gsM[m - 1]->beta_n_IFFT->out + s[m - 1],
467  gsM[m - 1]->beta_n_IFFT->out + s[m - 1] + 2 * s[m],
468  gsM[m]->beta_0_FFT->in);
469  }
470  alpha2Beta(gsM[m], s[m]);
471 
472  layer = log2(ceil((double)s[m + 1] / b));
473  GenStep(gsM[m]->alpha_n_IFFT->out + s[m], gsM[m]->beta_n_IFFT->out + s[m],
474  s[m + 1], layer);
475 
476  std::copy(gs[layer]->eta_0_IFFT->out, gs[layer]->eta_0_IFFT->out + s[m + 1],
477  gsM[m + 1]->eta_0_FFT->in);
478  std::copy(gs[layer]->xi_0_IFFT->out, gs[layer]->xi_0_IFFT->out + s[m + 1],
479  gsM[m + 1]->xi_0_FFT->in);
480  std::copy(gs[layer]->gamma, gs[layer]->gamma + s[m + 1], gsM[0]->gamma + n);
481  n += s[m + 1];
482  }
483 
484  n = s[s.size() - 1];
485  for (int m = s.size() - 2; m >= 0; --m) {
486  if (m == s.size() - 2) {
487  std::copy(gsM[m + 1]->xi_0_FFT->in, gsM[m + 1]->xi_0_FFT->in + n,
488  gsM[m]->xi_n_FFT->in);
489  std::copy(gsM[m + 1]->eta_0_FFT->in, gsM[m + 1]->eta_0_FFT->in + n,
490  gsM[m]->eta_n_FFT->in);
491  } else {
492  std::copy(gsM[m + 1]->xi_0_IFFT->out, gsM[m + 1]->xi_0_IFFT->out + n,
493  gsM[m]->xi_n_FFT->in);
494  std::copy(gsM[m + 1]->eta_0_IFFT->out, gsM[m + 1]->eta_0_IFFT->out + n,
495  gsM[m]->eta_n_FFT->in);
496  }
497  eta2Xi(gsM[m], s[m]);
498 
499  n += s[m];
500  }
501 }

510 inline void GSchurN::Compute(double* acf) {
511  for (int ii = 0; ii < N - 1; ++ii) {
512  alpha[ii] = -1 * acf[ii + 1];
513  beta[ii] = acf[ii];
514  }
515  GenMerge();
516  double sigma2 = log(acf[0]);
517  ldV = sigma2;
518  for (int ii = 0; ii < N - 1; ++ii) {
519  if (gsM[0]->gamma[ii] < 1) {
520  sigma2 += log(1 - gsM[0]->gamma[ii] * gsM[0]->gamma[ii]);
521  ldV += sigma2;
522  }
523  }
524  sigma2 = exp(sigma2);
525 
526  std::copy(gsM[0]->eta_0_IFFT->out, gsM[0]->eta_0_IFFT->out + N - 1, Phi);
527  Phi[N - 1] = 0;
528  Phi[0] /= sigma2;
529  for (int ii = 1; ii < N; ++ii) {
530  Phi[ii] += gsM[0]->xi_0_IFFT->out[ii - 1];
531  Phi[ii] /= sigma2;
532  }
533  return;
534 }
