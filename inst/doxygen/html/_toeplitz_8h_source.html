287 inline void Toeplitz::solve(double* y, const double* x) {
288  // Pointers to temporary storage: x_, x_fft_, y_, y_fft_, z_.
289  double* x_ = vec1_;
290  dcomplex* x_fft_ = vec1_fft_;
291  double* y_ = vec2_;
292  dcomplex* y_fft_ = vec2_fft_;
293  double* z_ = vec3_;
294  // dcomplex* z_fft_ = vec3_fft_;
295  if (!has_solve_) solve_setup();
296  if (N_ > 1) {
297  // GSchur algorithm only supports N > 1 case.
298  std::copy(x, x + N_, x_);
299  zero_fft(x_fft_, x_);
300  // y_ = L1' * x_
301  conv_fft(y_, tL1_fft_, x_fft_);
302  // z_ = L1 * y_ = L1 * L1' * x_
303  zero_fft(y_fft_, y_);
304  conv_fft(z_, L1_fft_, y_fft_);
305  // y_ = L2' * x_
306  conv_fft(y_, tL2_fft_, x_fft_);
307  // x_ = L2 * y_ = L2 * L2' * x_ (temporarily stored in x_)
308  zero_fft(y_fft_, y_);
309  conv_fft(x_, L2_fft_, y_fft_);
310  // y = (z_ - x_) / delta[1] = 1/delta[1] * (L1 * L1' * x_ - L2 * L2' * x_)
311  for (int ii = 0; ii < N_; ++ii) {
312  y[ii] = (z_[ii] - x_[ii]) / delta_[0];
313  }
314  } else {
315  // N = 1 case.
316  y[0] = x[0] / acf_[0];
317  }
318  return;
319 }

412 inline double Toeplitz::trace_deriv(const double* acf0) {
413  // Pointers to temporary storage: U1_, U1_fft_, U2_, U2_fft_, y_.
414  double* U1_ = vec1_;
415  dcomplex* U1_fft_ = vec1_fft_;
416  double* U2_ = vec2_;
417  dcomplex* U2_fft_ = vec2_fft_;
418  double* y_ = vec3_;
419  // dcomplex* y_fft_ = vec3_fft_;
420  double trace;
421  double acf00 = acf0[0];
422  if (N_ > 1) {
423  // GSchur algorithm only supports N > 1 case.
424  if (!has_solve_) solve_setup();
425  // check first term to avoid singularity
426  bool sng = fabs(acf0[0] < 0.0001);
427  if(sng) acf00 += 1.0;
428  std::copy(acf0, acf0 + N_, U1_);
429  if(sng) U1_[0] += 1.0;
430  zero_fft(U1_fft_, U1_);
431  std::fill(U2_, U2_ + 2 * N_, 0.0);
432  std::copy(acf0 + 1, acf0 + N_, U2_ + 1);
433  vfft_->fft(U2_fft_, U2_); // U2
434  // tr{U1'L1L1'U1}
435  conv_fft(y_, L1_fft_, U1_fft_);
436  trace = trace_LU(y_, y_);
437  // tr{U1'L2L2'U1}
438  conv_fft(y_, L2_fft_, U1_fft_);
439  trace -= trace_LU(y_, y_);
440  // tr{U2'L1L1'U2}
441  conv_fft(y_, L1_fft_, U2_fft_);
442  trace -= trace_LU(y_, y_);
443  // tr{U2'L2L2'U2}
444  conv_fft(y_, L2_fft_, U2_fft_);
445  trace += trace_LU(y_, y_);
446  // trace
447  trace /= delta_[0];
448  trace /= acf00;
449  if (sng) trace -= trace_inv(); // singularity correction
450  }
451  else {
452  trace = acf0[0] / acf_[0]; // N = 1 case.
453  }
454  return trace;
455 }

466 inline double Toeplitz::trace_hess(const double* acf1, const double* acf2) {
467  // Pointers to temporaries: phi_, x_, x_fft_, y_, y_fft_, z_, z_fft_, U1_, U2_
468  double* x_ = vec1_;
469  dcomplex* x_fft_ = vec1_fft_;
470  double* y_ = vec2_;
471  dcomplex* y_fft_ = vec2_fft_;
472  double* z_ = vec3_;
473  dcomplex* z_fft_ = vec3_fft_;
474  double* U1_ = vec4_;
475  // dcomplex* U1_fft_ = vec4_fft_;
476  double* U2_ = vec5_;
477  // dcomplex* U2_fft_ = vec5_fft_;
478  double* phi_ = vec6_;
479  double trace, kappa1, kappa2;
480  double acf20 = acf2[0];
481  if (N_ > 1) {
482  // GSchur algorithm only supports N > 1 case.
483  if (!has_solve_) solve_setup();
484  // check first term to avoid singularity
485  bool sng = fabs(acf2[0] < 0.0001);
486  if(sng) acf20 += 1.0;
487  // Store the negative derivative of delta in vector phi_, where phi_ = solve(acf_) * toep(acf1) * delta
488  product(phi_, delta_, acf1);
489  solve(phi_, phi_);
490  trace = trace_deriv(acf2);
491  if(sng) trace += trace_inv();
492  trace *= -phi_[0];
493  // kappa1
494  // U1_ = phi_ \conv acf2
495  std::copy(phi_, phi_ + N_, x_);
496  zero_fft(x_fft_, x_);
497  std::copy(acf2, acf2 + N_, z_);
498  if(sng) z_[0] += 1.0;
499  zero_fft(z_fft_, z_);
500  conv_fft(U1_, x_fft_, z_fft_);
501  // U2_ = delta_ \conv acf2
502  std::copy(delta_, delta_ + N_, y_);
503  zero_fft(y_fft_, y_);
504  conv_fft(U2_, y_fft_, z_fft_);
505  kappa1 = trace_LU(U1_, U2_) / acf20;
506  // U1_ = phi_ \conv zacf2, where zacf2 = acf2 with zero in first entry
507  z_[0] = 0.0;
508  vfft_->fft(z_fft_, z_);
509  conv_fft(U1_, x_fft_, z_fft_);
510  // U2_ = delta_ \conv zacf2
511  conv_fft(U2_, y_fft_, z_fft_);
512  kappa1 -= trace_LU(U1_, U2_) / acf20;
513  // kappa2
514  // U1_ = zrphi \conv acf2, where zrphi = phi with zero in first entry, then reversed
515  x_[0] = 0.0;
516  std::reverse(x_ + 1, x_ + N_);
517  vfft_->fft(x_fft_, x_);
518  z_[0] = acf20;
519  vfft_->fft(z_fft_, z_);
520  conv_fft(U1_, x_fft_, z_fft_);
521  // U2_ = zrdelta \conv acf2
522  y_[0] = 0.0;
523  std::reverse(y_ + 1, y_ + N_);
524  vfft_->fft(y_fft_, y_);
525  conv_fft(U2_, y_fft_, z_fft_);
526  kappa2 = trace_LU(U1_, U2_) / acf20;
527  // U1_ = zrphi \conv zacf2
528  z_[0] = 0.0;
529  vfft_->fft(z_fft_, z_);
530  conv_fft(U1_, x_fft_, z_fft_);
531  // U2_ = zrdelta \conv zacf2
532  conv_fft(U2_, y_fft_, z_fft_);
533  kappa2 -= trace_LU(U1_, U2_) / acf20;
534  // finalize trace calculation
535  trace += 2.0 * (kappa1 - kappa2);
536  trace /= delta_[0];
537  if (sng) {
538  // singularity correction
539  double t0 = 0.0;
540  for (int ii = 0; ii < N_; ii++) {
541  t0 += (N_ - 2 * ii) * delta_[ii] * phi_[ii];
542  }
543  trace -= 2.0 * t0 / delta_[0];
544  trace += trace_inv() * phi_[0] / delta_[0];
545  }
546  }
547  else {
548  trace = acf1[0] * acf2[0] / acf_[0] / acf_[0]; // N = 1 case.
549  }
550  return trace;
551 }
