267  Invfft->Ifft();

269

333 inline double Toeplitz::traceProd(double* acf0) {
334  double trace;
335  if (n > 1) {
336  // GSchur algorithm only supports N > 1 case.
337  bool is_small_acf0;
338  double t0;
339  if (all_zeros(acf0, n)) {
340  return 0.0;
341  }
342  if (!has_solve) {
343  solve_setup();
344  }
345  std::copy(acf0 + 1, acf0 + n, U1fft->in + 1);
346  is_small_acf0 = fabs(acf0[0]) < 0.0001;
347  t0 = is_small_acf0 ? acf0[0] + 1.0 : acf0[0];
348  U1fft->in[0] = t0;
349  U1fft->fft(); // U1
350  std::copy(acf0 + 1, acf0 + n, U2fft->in + 1);
351  U2fft->fft(); // U2
352 
353  // tr{U1'L1L1'U1}
354  vecConv(Invfft->in, L1fft->out, U1fft->out, 2 * (n / 2 + 1));
355  Invfft->Ifft();
356  trace = trace_LU(Invfft->out, Invfft->out, n);
357  // tr{U1'L2L2'U1}
358  vecConv(Invfft->in, L2fft->out, U1fft->out, 2 * (n / 2 + 1));
359  Invfft->Ifft();
360  trace -= trace_LU(Invfft->out, Invfft->out, n);
361  // tr{U2'L1L1'U2}
362  vecConv(Invfft->in, L1fft->out, U2fft->out, 2 * (n / 2 + 1));
363  Invfft->Ifft();
364  trace -= trace_LU(Invfft->out, Invfft->out, n);
365  // tr{U2'L2L2'U2}
366  vecConv(Invfft->in, L2fft->out, U2fft->out, 2 * (n / 2 + 1));
367  Invfft->Ifft();
368  trace += trace_LU(Invfft->out, Invfft->out, n);
369 
370  // trace
371  trace /= Gs->Phi[0];
372  trace /= t0;
373  if (is_small_acf0) {
374  t0 = 0.0;
375  for (int ii = 1; ii < n; ii++) {
376  t0 += (n - 2 * ii) * Gs->Phi[ii] * Gs->Phi[ii];
377  }
378  trace -= n * Gs->Phi[0] + t0 / Gs->Phi[0];
379  }
380  } else {
381  // N = 1 case.
382  trace = acf0[0] / acf[0];
383  }
384  return trace;
385 }

411 inline double Toeplitz::traceDeriv(double* acf1, double* acf2) {
412  double trace2;
413  if (n > 1) {
414  // GSchur algorithm only supports N > 1 case.
415  double t0;
416  bool is_small_acf2;
417  if (all_zeros(acf1, n) | all_zeros(acf2, n)) {
418  return 0.0;
419  }
420  if (!has_solve) {
421  solve_setup();
422  }
423  // check first term
424  is_small_acf2 = fabs(acf2[0]) < 0.0001;
425  t0 = is_small_acf2 ? acf2[0] + 1.0 : acf2[0];
426 
427  // (1) phi2 = - T^-1 * V_1 * phi:
428 
429  // phi2 = V_1 * phi
430  std::copy(acf1, acf1 + n, Lxfft->in);
431  std::copy(acf1 + 1, acf1 + n, Lxfft->in + n + 1);
432  std::reverse(Lxfft->in + n + 1, Lxfft->in + 2 * n);
433  Lxfft->fft();
434  std::fill(Lxfft->in + n, Lxfft->in + 2 * n,
435  0); // (readjust back to 0 assignment for latter half)
436  std::copy(Gs->Phi, Gs->Phi + n, xfft->in);
437  xfft->fft();
438  vecConv(Invfft->in, Lxfft->out, xfft->out, 2 * (n / 2 + 1));
439  Invfft->Ifft();
440  std::copy(Invfft->out, Invfft->out + n, phi2);
441  // phi2 = - T^-1 * phi2
442  std::copy(phi2, phi2 + n, xfft->in);
443  xfft->fft();
444 
445  // L1 * L1' * x
446  vecConv(Invfft->in, L11fft->out, xfft->out, 2 * (n / 2 + 1));
447  Invfft->Ifft();
448  std::copy(Invfft->out, Invfft->out + n, Lxfft->in);
449  Lxfft->fft();
450  vecConv(Invfft->in, L1fft->out, Lxfft->out, 2 * (n / 2 + 1));
451  Invfft->Ifft();
452  std::copy(Invfft->out, Invfft->out + n, phi2);
453 
454  // L2 * L2' * x
455  vecConv(Invfft->in, L22fft->out, xfft->out, 2 * (n / 2 + 1));
456  Invfft->Ifft();
457  std::copy(Invfft->out, Invfft->out + n, Lxfft->in);
458  Lxfft->fft();
459  vecConv(Invfft->in, L2fft->out, Lxfft->out, 2 * (n / 2 + 1));
460  Invfft->Ifft();
461 
462  // phi2 = -1/phi[1] * (L1 L1'x - L2 L2'x)
463  for (int ii = 0; ii < n; ++ii) {
464  phi2[ii] -= Invfft->out[ii];
465  phi2[ii] /= Gs->Phi[0];
466  }
467 
468  // tr = phi2[1] * tr(T^-1 * V_2) + 2 * tr(L1(-phi2) * L1(phi)' * V_2) - 2 *
469  // tr(_L2(-phi2) * _L2(phi)' * V_2)
470 
471  // (1) phi2[1] * tr(T^-1 * V_2)
472  trace2 = phi2[0] * traceProd(acf2);
473 
474  // (2) tr(L1(-phi2) * L1(phi)' * V_2) = -tr(L1(phi2) * L1(phi)' * L1'(acf2)
475  // * L1(acf2)) + tr(L1(phi2) * L1(phi)' * L2'(acf2) * L2(acf2))
476 
477  // (2.1) tr(L1(phi2) * L1(phi)' * L1'(acf2) * L1(acf2)) = trace_LU(L1(acf2)
478  // * L1(phi2), L1(acf2) * L1(phi)) = temVec
479  std::copy(acf2, acf2 + n, U1fft->in);
480  U1fft->in[0] = t0;
481  U1fft->fft();
482  std::copy(phi2, phi2 + n, Lxfft->in);
483  Lxfft->fft();
484  vecConv(Invfft->in, U1fft->out, Lxfft->out, 2 * (n / 2 + 1));
485  Invfft->Ifft();
486  std::copy(Invfft->out, Invfft->out + n, temVec);
487  std::copy(Gs->Phi, Gs->Phi + n, xfft->in);
488  xfft->fft();
489  vecConv(Invfft->in, U1fft->out, xfft->out, 2 * (n / 2 + 1));
490  Invfft->Ifft();
491  trace2 -= 2 * trace_LU(Invfft->out, temVec, n) / t0;
492 
493  // (2.2) tr(L1(phi2) * L1(phi)' * L2'(acf2) * L2(acf2)) = trace_LU(L2(acf2)
494  // * L1(phi2), L2(acf2) * L1(phi)) = temVec
495  std::copy(acf2 + 1, acf2 + n, U2fft->in + 1);
496  U2fft->in[0] = 0;
497  U2fft->fft();
498  vecConv(Invfft->in, U2fft->out, Lxfft->out, 2 * (n / 2 + 1));
499  Invfft->Ifft();
500  std::copy(Invfft->out, Invfft->out + n, temVec);
501  vecConv(Invfft->in, U2fft->out, xfft->out, 2 * (n / 2 + 1));
502  Invfft->Ifft();
503  trace2 += 2 * trace_LU(Invfft->out, temVec, n) / t0;
504 
505  // (3) tr(_L2(phi2) * _L2(phi)' * Toeplitz_3) = tr(_L2(phi2) * _L2'(phi) *
506  // L1'(acf2) * L1(acf2)) / acf2[1] - tr(_L2(phi2) * _L2(phi)' * L2'(acf2) *
507  // L2(acf2)) / acf2[1]
508 
509  // (3.1) tr(L2(phi2) * L2'(phi) * L1'(acf2) * L1(acf2)) = trace_LU(L1(acf2)
510  // * _L2(phi2), L1(acf2) * _L2(phi)) = temVec
511  Lxfft->in[0] = 0;
512  std::copy(phi2 + 1, phi2 + n, Lxfft->in + 1);
513  std::reverse(Lxfft->in + 1, Lxfft->in + n);
514  Lxfft->fft();
515  vecConv(Invfft->in, U1fft->out, Lxfft->out, 2 * (n / 2 + 1));
516  Invfft->Ifft();
517  std::copy(Invfft->out, Invfft->out + n, temVec);
518  xfft->in[0] = 0;
519  std::copy(Gs->Phi + 1, Gs->Phi + n, xfft->in + 1);
520  std::reverse(xfft->in + 1, xfft->in + n);
521  xfft->fft();
522  vecConv(Invfft->in, U1fft->out, xfft->out, 2 * (n / 2 + 1));
523  Invfft->Ifft();
524  trace2 += 2 * trace_LU(Invfft->out, temVec, n) / t0;
525

527  // trace_LU(L2(acf2) * _L2(phi2), L2(acf2) * _L2(phi)) = temVec
528  vecConv(Invfft->in, U2fft->out, Lxfft->out, 2 * (n / 2 + 1));
529  Invfft->Ifft();
530  std::copy(Invfft->out, Invfft->out + n, temVec);
531  vecConv(Invfft->in, U2fft->out, xfft->out, 2 * (n / 2 + 1));
532  Invfft->Ifft();
533  trace2 -= 2 * trace_LU(Invfft->out, temVec, n) / t0;
534 
535  // trace2 = trace2 / phi[1]
536  trace2 /= Gs->Phi[0];
537  trace2 *= -1.0;
538  if (is_small_acf2) {
539  t0 = 0.0;
540  for (int ii = 1; ii < n; ii++) {
541  t0 += (n - 2 * ii) * Gs->Phi[ii] * phi2[ii];
542  }
543  trace2 -= 2 * n * phi2[0] + 2 * t0 / Gs->Phi[0];
544  }
545  } else {
546  // N = 1 case.
547  trace2 = acf1[0] * acf2[0] / acf[0] / acf[0];
548  }
